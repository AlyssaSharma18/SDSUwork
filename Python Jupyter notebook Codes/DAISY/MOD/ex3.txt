WRITE "A THREE STATE MODEL"$

% B_ IS THE VARIABLE VECTOR 
B_:={u1,y1,x1,x2,x3}$

FOR EACH EL_ IN B_ DO DEPEND EL_,T$

%B1_ IS THE UNKNOWN PARAMETER VECTOR
B1_:={p1,p2,p3,p4,p5,p6,p7}$

%NUMBER OF STATES 
NX_:=3$
%NUMBER OF INPUTS 
NU_:=1$
%NUMBER OF OUTPUTS 
NY_:=1$

%MODEL EQUATIONS
C_:={df(x1,t)=-p1*x1+p2*x2+u1,
     df(x2,t)=p3*x1-p4*x2+p5*x3,
     df(x3,t)=p6*x1-p7*x3,
     x1=y1}$
 
SEED_:=25$
DAISY()$
END$
